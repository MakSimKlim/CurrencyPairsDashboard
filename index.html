// ИСПРАВЛЕНО: Полная блокировка свайпа. Перетаскивание активируется СТРОГО после 500мс удержания function setupTouchHoldLogic(card) { let holdTimer = null; const HOLD_DURATION = 500; // Время удержания в мс function startHold(e) { // Если нажали на кнопку удаления, игнорируем if (e.target.closest('button')) return; cleanHold(); holdTimer = setTimeout(() => { // 1. Разрешаем перетаскивание для браузера card.setAttribute('draggable', 'true'); card.classList.add('hold-active'); // 2. Виброотклик if (navigator.vibrate) navigator.vibrate(40); // 3. ТРИГГЕР ДЛЯ ПОЛИФИЛА: Принудительно заставляем DragDropTouch думать, // что dragstart начался прямо сейчас в этой точке, игнорируя предыдущий свайп const touch = e.touches ? e.touches[0] : e; // Очищаем внутренний триггер полифила, если он пытался запуститься раньше времени if (window.DragDropTouch && window.DragDropTouch._instance) { window.DragDropTouch._instance._dragSource = null; } const dragStartEvent = new DragEvent('dragstart', { bubbles: true, cancelable: true, clientX: touch.clientX, clientY: touch.clientY }); card.dispatchEvent(dragStartEvent); }, HOLD_DURATION); } function cleanHold() { if (holdTimer) { clearTimeout(holdTimer); holdTimer = null; } } function resetCardState(e) { // Если таймер не отработал (был просто быстрый свайп или клик) — отменяем всё if (!card.classList.contains('hold-active')) { cleanHold(); card.setAttribute('draggable', 'false'); } } // Слушатели тач-событий card.addEventListener('touchstart', startHold, { passive: true }); card.addEventListener('touchmove', (e) => { // Если палец двинулся ДО истечения 500мс — это обычный скролл страницы. // Сбрасываем таймер, чтобы карточка не захватилась в процессе скролла. if (!card.classList.contains('hold-active')) { cleanHold(); } }, { passive: true }); card.addEventListener('touchend', resetCardState); card.addEventListener('touchcancel', resetCardState); // Поведение для десктопных мышей (без задержки) card.addEventListener('mousedown', (e) => { if (e.target.closest('button')) return; card.setAttribute('draggable', 'true'); }); card.addEventListener('mouseup', () => { if (!draggedCurrencyCode) card.setAttribute('draggable', 'false'); }); } function setupDragAndDropEvents(card) { card.addEventListener('dragstart', () => { draggedCurrencyCode = card.getAttribute('data-currency'); card.classList.add('opacity-40', 'border-emerald-500'); }); card.addEventListener('dragend', () => { draggedCurrencyCode = null; // ИСПРАВЛЕНО: После окончания переноса жестко возвращаем всем карточкам исходное состояние document.querySelectorAll('.currency-card').forEach(el => { el.setAttribute('draggable', 'false'); el.classList.remove('opacity-40', 'border-emerald-500', 'hold-active', 'translate-y-1'); }); }); card.addEventListener('dragover', (e) => { e.preventDefault(); const targetCurrency = card.getAttribute('data-currency'); if (targetCurrency !== draggedCurrencyCode) card.classList.add('border-emerald-500', 'translate-y-1'); }); card.addEventListener('dragleave', () => card.classList.remove('border-emerald-500', 'translate-y-1')); card.addEventListener('drop', (e) => { e.preventDefault(); const targetCurrency = card.getAttribute('data-currency'); if (!draggedCurrencyCode || draggedCurrencyCode === targetCurrency) return; const draggedIndex = userCurrencies.indexOf(draggedCurrencyCode); const targetIndex = userCurrencies.indexOf(targetCurrency); if (draggedIndex !== -1 && targetIndex !== -1) { userCurrencies.splice(draggedIndex, 1); userCurrencies.splice(targetIndex, 0, draggedCurrencyCode); saveToLocalStorage(); updateDashboard(); } }); }
